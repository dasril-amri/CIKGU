Panduan Google Spreadsheet

Ikuti beberapa langkah mudah berikut untuk menghubungkan absensi dengan spreadsheet Guru secara real-time:

  1. Buka lembar kerja Google Sheets Anda atau gunakan link referensi ini: Akses Link Spreadsheet.
  2. Klik menu Ekstensi (Extensions) → Apps Script.
  3. Hapus skrip bawaan yang ada di editor, lalu paste kode berikut:
function doPost(e) {
  try {
    var payload = JSON.parse(e.postData.contents);
    var type = payload.type; // "harian", "rekap", atau "nilai"
    var data = payload.data;
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    
    var sheetName;
    if (type === "harian") sheetName = "Absensi Harian";
    else if (type === "rekap") sheetName = "Rekap Bulanan";
    else if (type === "nilai") sheetName = "Penilaian Siswa";
    else sheetName = "Lainnya";
    
    var sheet = ss.getSheetByName(sheetName);
    if (!sheet) {
      sheet = ss.insertSheet(sheetName);
      if (type === "harian") {
        sheet.appendRow(["Tanggal", "Kelas", "Nama Siswa", "Status (H/S/I/A/TS)", "Keterangan Tambahan"]);
      } else if (type === "rekap") {
        sheet.appendRow(["Kelas", "Bulan", "Nama Siswa", "Hadir (H)", "Sakit (S)", "Izin (I)", "Alfa (A)", "Terlambat (TS)"]);
      } else if (type === "nilai") {
        sheet.appendRow(["Kelas", "Nama Siswa", "Rata-Rata Praktik", "Rata-Rata Pengetahuan", "Quiz (Q1)", "UTS", "Nilai Akhir"]);
      }
    }
    
    data.forEach(function(row) {
      if (type === "harian") {
        sheet.appendRow([row.tanggal, row.kelas, row.nama, row.status, row.keterangan]);
      } else if (type === "rekap") {
        sheet.appendRow([row.kelas, row.bulan, row.nama, row.hadir, row.sakit, row.izin, row.alfa, row.terlambat]);
      } else if (type === "nilai") {
        sheet.appendRow([row.kelas, row.nama, row.praktik_avg, row.pengetahuan_avg, row.q1, row.uts, row.nilai_akhir]);
      }
    });
    
    return ContentService.createTextOutput(JSON.stringify({status: "success", message: "Data berhasil dimasukkan ke " + sheetName}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({status: "error", message: error.toString()}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
  1. Klik tombol **Simpan (ikon Disket)** di bagian atas Apps Script.
  2. Klik tombol **Terapkan (Deploy)** → **Terapkan Baru (New deployment)**.
  3. Konfigurasikan Deployment:
    • Jenis Deployment: Aplikasi Web (Web App)
    • Jalankan sebagai (Execute as): Saya (Me / email guru)
    • Siapa yang memiliki akses (Who has access): Siapa saja (Anyone)
  4. Klik **Terapkan (Deploy)** dan setujui izin akses (*Authorize Access*) dari akun Google Anda.
  5. Salin tautan **URL Aplikasi Web (Web App URL)** yang ditampilkan.
  6. Kembali ke aplikasi ini, buka **Pengaturan PIN**, tempel URL tersebut di kolom yang disediakan dan simpan.
